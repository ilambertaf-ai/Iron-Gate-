  <button onclick="saveDay()">Save</button>

<script>
  const calendarEl = document.getElementById('calendar');
  const statsEl = document.getElementById('stats');
  let selectedDate = null;

  function loadCalendar() {
    calendarEl.innerHTML = '';
    const year = new Date().getFullYear();

    for (let i = 1; i <= 365; i++) {
      const day = document.createElement('div');
      day.className = 'day';
      day.textContent = `Day ${i}`;
      day.onclick = () => selectDay(i);
      calendarEl.appendChild(day);
    }
  }

  function selectDay(dayIndex) {
    selectedDate = dayIndex;
    document.querySelectorAll('.day').forEach(d => d.classList.remove('active'));
    document.querySelectorAll('.day')[dayIndex - 1].classList.add('active');

    const data = JSON.parse(localStorage.getItem(`day-${dayIndex}`)) || {};
    document.getElementById('minutes').value = data.minutes || '';
    document.getElementById('sets').value = data.sets || '';
    document.getElementById('reps').value = data.reps || '';
  }

  function saveDay() {
    if (!selectedDate) return;

    const data = {
      minutes: document.getElementById('minutes').value,
      sets: document.getElementById('sets').value,
      reps: document.getElementById('reps').value,
    };

    localStorage.setItem(`day-${selectedDate}`, JSON.stringify(data));
    alert('Saved');
  }

  loadCalendar();
</script>
